365  List<TimedEvent> CopyEvents(IEnumerable<TimedEvent> source, IEnumerable<MonoBehaviour> sourceComponents, IEnumerable<MonoBehaviour> destinationComponents)
366  {
367  var output = new List<TimedEvent>();
368  foreach (var timedEvent in source)
369  {
370  var newTimedEvent = new TimedEvent { Duration = timedEvent.Duration, Event = new UnityEngine.Events.UnityEvent() };
371  for (var j = 0; j < timedEvent.Event.GetPersistentEventCount(); j++)
372  {
373  var target = (MonoBehaviour)timedEvent.Event.GetPersistentTarget(j);
374  if (sourceComponents.Contains(target))
375  {
376  target = destinationComponents.GetComponentForName(target.gameObject.name, target.GetType());
377  }
378 
379  var methodName = timedEvent.Event.GetPersistentMethodName(j);
380  newTimedEvent.Event.AddListener(new UnityEngine.Events.UnityAction(() => target.Invoke(methodName, 0f)));
381  }
382 
383  output.Add(newTimedEvent);
384  }
385  return output;
