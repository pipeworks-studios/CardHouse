470  {
471  var newMountedCards = new List<Card>();
472  while (MountedCards.Count > 0)
473  {

475  newMountedCards.Add(MountedCards[chosenIndex]);
476  MountedCards.RemoveAt(chosenIndex);
477  }
478  MountedCards = newMountedCards;

480  var seekerSetList = new SeekerSetList();
481  if (isInstant)

484  }
485  else
486  {
487  foreach (var card in MountedCards)
488  {

494  {
495  if (card.CanBeUpsideDown)
